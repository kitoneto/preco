    <script>
        function openTab(tabName) {
            const tabs = document.getElementsByClassName('tab');
            for (let i = 0; i < tabs.length; i++) {
                tabs[i].classList.remove('active');
            }
            
            const tabContents = document.getElementsByClassName('tab-content');
            for (let i = 0; i < tabContents.length; i++) {
                tabContents[i].classList.remove('active');
            }
            
            event.currentTarget.classList.add('active');
            document.getElementById(tabName).classList.add('active');
        }
        
        function showModal(modalId) {
            document.getElementById(modalId).style.display = 'block';
        }
        
        function hideModal(modalId) {
            document.getElementById(modalId).style.display = 'none';
        }
        
        function adicionarItem() {
            const table = document.getElementById('itens-orcamento');
            const newRow = table.insertRow();
            
            newRow.innerHTML = `
                <td>
                    <select>
                        <option>Levantamento Topográfico</option>
                        <option>Projeto Elétrico</option>
                        <option>Mapeamento GIS</option>
                    </select>
                </td>
                <td><input type="number" value="1" min="1" step="1"></td>
                <td>600,00 €</td>
                <td>600,00 €</td>
                <td><button class="btn" type="button" onclick="removerItem(this)">Remover</button></td>
            `;
        }
        
        function removerItem(button) {
            const row = button.parentNode.parentNode;
            row.parentNode.removeChild(row);
        }
        
        // Fechar modal ao clicar fora
        window.onclick = function(event) {
            if (event.target.className === 'modal') {
                event.target.style.display = 'none';
            }
        }
        
        // Calcular preço de venda automático
        document.getElementById('custo-servico').addEventListener('input', calcularPreco);
        document.getElementById('margem-servico').addEventListener('input', calcularPreco);
        
        function calcularPreco() {
            const custo = parseFloat(document.getElementById('custo-servico').value) || 0;
            const margem = parseFloat(document.getElementById('margem-servico').value) || 0;
            const preco = custo * (1 + margem/100);
            document.getElementById('preco-venda').textContent = preco.toFixed(2) + ' €';
        }
